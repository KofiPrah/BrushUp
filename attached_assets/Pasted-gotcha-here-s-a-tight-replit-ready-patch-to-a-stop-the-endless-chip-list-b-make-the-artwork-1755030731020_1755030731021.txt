# 3) JS: limit chips to a small set, add “Show more”, wire bigger image links

**In your feed JS**, add/replace these helpers:

```js
function renderChipsLimited(list, containerId, moreBtnId, side){
  const wrap = document.getElementById(containerId);
  const moreBtn = document.getElementById(moreBtnId);
  const MAX = 12;                       // show only 12 by default
  let expanded = false;

  function render(){
    wrap.innerHTML = "";
    const items = expanded ? list : list.slice(0, MAX);
    items.forEach(t => wrap.appendChild(chip(t, side)));
    wrap.classList.toggle("is-expanded", expanded);
    // fade overlay only when collapsed and overflowed
    wrap.nextElementSibling.style.display = (!expanded && list.length > MAX) ? "block" : "none";
    moreBtn.style.display = (list.length > MAX) ? "inline-flex" : "none";
    moreBtn.textContent = expanded ? "Show less" : `Show ${list.length - MAX} more`;
  }

  moreBtn.onclick = () => { expanded = !expanded; render(); };
  render();
}

// update your renderPair:
function renderPair(p){
  state.pair = p;
  const s = p.spotlight, c = p.counterpoint;
  document.getElementById('spotlight-img').src   = s.image_url || '';
  document.getElementById('counter-img').src     = c.image_url || '';
  document.getElementById('spotlight-link').href = `/artworks/${s.id}/`;
  document.getElementById('counter-link').href   = `/artworks/${c.id}/`;

  renderChipsLimited(p.chips.pro.map(t=>({...t, polarity:'PRO'})), "pro-chips", "pro-more", "spotlight");
  renderChipsLimited(p.chips.con.map(t=>({...t, polarity:'CON'})), "con-chips", "con-more", "spotlight");
}
```

…and keep your existing `chip()` function, but make sure it accepts a `side`:

```js
function chip(t, side){
  const pol = t.polarity === 'PRO' ? '-pro' : '-con';
  const b = h('button', {class:`chip ${pol}`, 'data-id':t.id});
  b.textContent = (t.polarity==='PRO'?'+ ':'• ') + t.label;
  b.onclick = ()=>{
    addTag(side, t.id);
    b.classList.toggle('is-selected');
    document.getElementById('next').disabled = !hasAnyTag() && !document.getElementById('quick-note').value.trim();
  };
  return b;
}
```
